void traverse(Graph &G){

}
//递归算法
void DFS(Graph &G, int id){
    //访问
    visit[id] = 1;
    edge = G.node[id].first;
    while(edge){
        if(visit[edge.to_node]==0) DFS(G,edge.to_node);
        edge=edge.next;
    }
}
